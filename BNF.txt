<atribuição>      :=   <id> "=" <expr_arit> ";"
<expr_arit>       :=   <expr_arit> "+" <termo>   | <expr_arit> "-" <termo> | <termo>
<termo>           :=   <termo> "*" <fator> | <termo> “/” <fator> | <fator>
<fator>           :=   “(“ <expr_arit> “)” | <id> | <float> | <inteiro> | <char>
